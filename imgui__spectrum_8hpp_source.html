5/*
6Color definitions in ImGui are a good starting point,
7but do not cover all the intricacies of Spectrum's possible colors
8in controls and widgets.
9
10One big difference is that ImGui communicates widget activity
11(hover, pressed) with their background, while spectrum uses a mix
12of background and border, with border being the most common choice.
13
14Because of this, we reference extra colors in spectrum from
15imgui.cpp and imgui_widgets.cpp directly, and to make that work,
16we need to have them defined at here at compile time.
17*/

41 unsigned int Color(unsigned int c) {
42 // add alpha.
43 // also swap red and blue channel for some reason.
44 // todo: figure out why, and fix it.
45 const short a = 0xFF;
46 const short r = (c >> 16) & 0xFF;
47 const short g = (c >> 8) & 0xFF;
48 const short b = (c >> 0) & 0xFF;
49 return(a << 24)
50 | (r << 0)
51 | (g << 8)
52 | (b << 16);
53 }
54 }

57 inline unsigned int color_alpha(unsigned int alpha, unsigned int c) {
58 return ((alpha & 0xFF) << 24) | (c & 0x00FFFFFF);
59 }
