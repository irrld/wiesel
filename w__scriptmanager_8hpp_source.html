38class FieldData {
39 public:
40 FieldData(MonoClassField* field, const std::string& fieldName,
41 uint32_t fieldFlags) : m_Field(field),
42 m_FieldName(fieldName),
43 m_FieldFlags(fieldFlags) {
44 std::string typeName = mono_type_get_name(mono_field_get_type(m_Field));
45 if (typeName == "System.Boolean") {
46 m_FieldType = FieldType::Boolean;
47 } else if (typeName == "System.Single") {
48 m_FieldType = FieldType::Float;
49 } else if (typeName == "System.Double") {
50 m_FieldType = FieldType::Double;
51 } else if (typeName == "System.Int32") {
52 m_FieldType = FieldType::Integer;
53 } else if (typeName == "System.Int64") {
54 m_FieldType = FieldType::Long;
55 } else if (typeName == "System.UInt32") {
56 m_FieldType = FieldType::UnsignedInteger;
57 } else if (typeName == "System.UInt64") {
58 m_FieldType = FieldType::UnsignedLong;
59 } else if (typeName == "System.String") {
60 m_FieldType = FieldType::String;
61 } else {
62 m_FieldType = FieldType::Object;
63 }
64 m_FormattedName = FormatVariableName(m_FieldName);
65 }
66
67 template <typename T>
68 void Set(MonoObject* instance, T value) {
69 mono_field_set_value(instance, m_Field, value);
70 }
71
72 template <typename T>
73 T Get(MonoObject* instance) {
74 T value;
75 mono_field_get_value(instance, m_Field, &value);
76 return value;
77 }
78
79 MonoClassField* GetField() const { return m_Field; }
80 const std::string& GetFieldName() const { return m_FieldName; }
81 uint32_t GetFieldFlags() const { return m_FieldFlags; }
82 FieldType GetFieldType() const { return m_FieldType; }
83 const std::string& GetFormattedName() const { return m_FormattedName; }
84
85 private:
86 MonoClassField* m_Field;
87 std::string m_FieldName;
88 std::string m_FormattedName;
89 uint32_t m_FieldFlags;
90 FieldType m_FieldType;
91};
