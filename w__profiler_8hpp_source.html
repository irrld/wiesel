52template <size_t N>
53struct ChangeResult {

55};

57template <size_t N, size_t K>
58constexpr auto CleanupOutputString(const char (&expr)[N],
59 const char (&remove)[K]) {
60 ChangeResult<N> result = {};
61
62 size_t srcIndex = 0;
63 size_t dstIndex = 0;
64 while (srcIndex < N) {
65 size_t matchIndex = 0;
66 while (matchIndex < K - 1 && srcIndex + matchIndex < N - 1 &&
67 expr[srcIndex + matchIndex] == remove[matchIndex])
68 matchIndex++;
69 if (matchIndex == K - 1)
70 srcIndex += matchIndex;
71 result.Data[dstIndex++] = expr[srcIndex] == '"' ? '\'' : expr[srcIndex];
72 srcIndex++;
73 }
74 return result;
75}
