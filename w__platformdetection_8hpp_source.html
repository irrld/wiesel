14// https://github.com/TheCherno/Hazel
15// Platform detection using predefined macros
16#ifdef _WIN32
17/* Windows x64/x86 */
18#ifdef _WIN64
19/* Windows x64 */
20#define WIESEL_PLATFORM_WINDOWS
21#else
22/* Windows x86 */
23#error "x86 Builds are not supported!"
24#endif
25#elif defined(__APPLE__) || defined(__MACH__)
26#include <TargetConditionals.h>
27/* TARGET_OS_MAC exists on all the platforms
28 * so we must check all of them (in this order)
29 * to ensure that we're running on MAC
30 * and not some other Apple platform */
31#if TARGET_IPHONE_SIMULATOR == 1
32#error "IOS simulator is not supported!"
33#elif TARGET_OS_IPHONE == 1
34#define WIESEL_PLATFORM_IOS
35#error "IOS is not supported!"
36#elif TARGET_OS_MAC == 1
37#define WIESEL_PLATFORM_MACOS
38#else
39#error "Unknown Apple platform!"
40#endif
41/* We also have to check __ANDROID__ before __linux__
42 * since android is based on the linux kernel
43 * it has __linux__ defined */
44#elif defined(__ANDROID__)
45#define WIESEL_PLATFORM_ANDROID
46#error "Android is not supported!"
47#elif defined(__linux__)
48#define WIESEL_PLATFORM_LINUX
49#error "Linux is not supported!"
50#else
51/* Unknown compiler/platform */
52#error "Unknown platform!"
53#endif // End of platform detection
