21// Resolve which function signature macro will be used. Note that this only
22// is resolved when the (pre)compiler starts, so the syntax highlighting
23// could mark the wrong one in your editor!
24#if defined(__GNUC__) || (defined(__MWERKS__) && (__MWERKS__ >= 0x3000)) || \
25 (defined(__ICC) && (__ICC >= 600)) || defined(__ghs__)
26#define WIESEL_FUNC_SIG __PRETTY_FUNCTION__
27#elif defined(__DMC__) && (__DMC__ >= 0x810)
28WIESEL_FUNC_SIG
29#define WIESEL_FUNC_SIG __PRETTY_FUNCTION__
30#elif (defined(__FUNCSIG__) || (_MSC_VER))
31#define WIESEL_FUNC_SIG __FUNCSIG__
32#elif (defined(__INTEL_COMPILER) && (__INTEL_COMPILER >= 600)) || \
33 (defined(__IBMCPP__) && (__IBMCPP__ >= 500))
34#define WIESEL_FUNC_SIG __FUNCTION__
35#elif defined(__BORLANDC__) && (__BORLANDC__ >= 0x550)
36#define WIESEL_FUNC_SIG __FUNC__
37#elif defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 199901)
38#define WIESEL_FUNC_SIG __func__
39#elif defined(__cplusplus) && (__cplusplus >= 201103)
40#define WIESEL_FUNC_SIG __func__
41#else
42#define WIESEL_FUNC_SIG "WIESEL_FUNC_SIG unknown!"
43#endif
